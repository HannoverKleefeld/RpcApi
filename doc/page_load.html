Standard loading

$api=new RpcApi('Sonos [PLAY_1].json',new RpcLogger());


Import device from XML

\\ Sony TV
$api=new RpcApi('http://192.168.112.61:8080/description.xml',new RpcLogger());
\\ Fritz!box
$api=new RpcApi('http://fritz.box:49000/tr64desc.xml',new RpcLogger());
$api=new RpcApi('http://fb_username:fb_password@fritz.box:49000/tr64desc.xml',new RpcLogger());
\\ Synology nas
$api=new RpcApi('http://nas:55480/ssdp/desc-DSM-eth0.xml',new RpcLogger());
\\ Dreambox
$api=new RpcApi('http://192.168.112.65:8200/rootDesc.xml',new RpcLogger());
\\ Sonos
$api=new RpcApi('http://192.168.112.54:1400/xml/device_description.xml',new RpcLogger());
\\ Samsung AVR
$api=new RpcApi('http://192.168.112.60:7676/smp_16_',new RpcLogger());
\\ Plex
$api=new RpcApi('http://192.168.112.1:32469/DeviceDescription.xml',new RpcLogger());

Build device from rpc.inject

\\ Homematic CCU
$api=new RpcApi('build://homematic',new RpcLogger());
\\ IP-Symcon
api=new RpcApi('build://ips_user_name:ips_password@ips_ip:ips_port/ipsymcon',new RpcLogger());
